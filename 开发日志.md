#### 任务 1：API 客户端服务
**状态**：✅ 已完成 (2026-04-03)

- [x] `services/api_client.py` - 基于 httpx 的异步 HTTP 客户端（OpenAI 兼容格式）
- [x] `services/models.py` - Pydantic 请求/响应模型（OpenAI 兼容格式）
- [x] 支持 Chat Completions API (`POST /v1/chat/completions`)
- [x] 流式和非流式两种模式
- [x] 错误处理与重试逻辑（429 退避、5xx 重试）
- [x] 自定义异常类（OpenAIAuthError/OpenAIRateLimitError 等）

**API 使用示例**：
```python
from mini_claude.services import OpenAIClient

# 非流式请求
async with OpenAIClient() as client:
    response = await client.create_chat_completion(
        model="claude-3-opus-20240229",
        messages=[{"role": "user", "content": "Hello!"}],
        max_completion_tokens=1024,
    )
    print(response.choices[0].message.content)

# 流式请求
async with OpenAIClient() as client:
    async for chunk in client.create_chat_completion_stream(
        model="claude-3-opus-20240229",
        messages=[{"role": "user", "content": "Hello!"}],
        max_completion_tokens=1024,
    ):
        if chunk.choices:
            delta = chunk.choices[0].delta
            if delta.content:
                print(delta.content, end="")
```

### 完整对话流程（含工具调用）
```python
import asyncio
from mini_claude.services import (
    Conversation,
    OpenAIClient,
    ToolDefinition,
    FunctionDefinition,
    FunctionParameters,
)

async def chat_with_history():
    # 1. 创建对话（带系统提示）
    conv = Conversation(
        system_prompt="你是一个 AI 编程助手，擅长使用工具完成编程任务。",
        model="doubao-seed-2-0-pro-260215",
    )

    # 2. 添加历史对话（从之前的交互）
    conv.add_user_message("请帮我查看当前目录下的文件")

    # 3. 定义可用的工具
    tools = [
        ToolDefinition(
            type="function",
            function=FunctionDefinition(
                name="BashTool",
                description="执行 shell 命令",
                parameters=FunctionParameters(
                    properties={
                        "command": {"type": "string", "description": "要执行的命令"},
                        "description": {"type": "string", "description": "命令描述"},
                    },
                    required=["command"],
                ),
            ),
        )
    ]

    # 4. 发送请求给 LLM
    async with OpenAIClient() as client:
        response = await client.create_chat_completion(
            model=conv.model or "doubao-seed-2-0-pro-260215",
            messages=conv.get_messages_for_api(),
            tools=tools,
        )

        # 5. 处理响应
        message = response.choices[0].message
        if message.tool_calls:
            # LLM 请求调用工具
            for tool_call in message.tool_calls:
                # 记录工具调用到对话
                conv.add_assistant_message([
                    {
                        "type": "tool_call",
                        "id": tool_call.id,
                        "name": tool_call.function.name,
                        "arguments": {"command": "ls -la"},
                    }
                ])

                # 执行工具（这里简化处理）
                tool_result = "file1.py\nfile2.txt\nREADME.md"

                # 添加工具结果到对话
                conv.add_tool_result(
                    tool_call_id=tool_call.id,
                    content=tool_result,
                )

            # 6. 继续对话（发送工具结果）
            response2 = await client.create_chat_completion(
                model=conv.model or "doubao-seed-2-0-pro-260215",
                messages=conv.get_messages_for_api(),
                tools=tools,
            )

            # 添加最终回复
            conv.add_assistant_message(response2.choices[0].message.content or "")

        else:
            # 普通文本回复
            conv.add_assistant_message(message.content or "")

    # 7. 保存对话历史
    manager = get_conversation_manager()
    manager.save(conv)

    print(f"对话 ID: {conv.id}")
    print(f"Token 估算: {conv.estimate_tokens()}")

# 运行示例
asyncio.run(chat_with_history())
```

### 对话管理
```python
manager = get_conversation_manager()

# 列出所有对话
conversations = manager.list()
for conv in conversations:
    print(f"ID: {conv['id']}")
    print(f"最后活跃: {conv['updated_at']}")
    print(f"内容: {conv['snippet']}")

# 加载特定对话
old_conv = manager.load("some-conversation-id")
if old_conv:
    # 继续之前的对话
    old_conv.add_user_message("继续刚才的话题...")
```
